בָּאָ֣ה מִן־הַ*צָּפ֗וֹן* עָנָ֤ן גָּדוֹל֙ וְ

וְהִנֵּ֤ה מִ*צָּפוֹן֙* לְשַׁ֣עַר הַ

בָּשָׂ֖ר מִנֶּ֥גֶב *צָפֽוֹן* וְיָֽדְעוּ֙ כָּל־

מֶֽלֶךְ־בָּבֶ֛ל מִ*צָּפ֖וֹן* מֶ֣לֶךְ מְלָכִ֑ים בְּ

שָׁ֣מָה נְסִיכֵ֥י *צָפ֛וֹן* כֻּלָ֖ם וְ

בֵּ֚ית תּֽוֹגַרְמָ֔ה יַרְכְּתֵ֥י *צָפ֖וֹן* וְאֶת־כָּל

ְךָ֙ מִיַּרְכְּתֵ֣י *צָפ֔וֹן* אַתָּ֕ה וְעַמִּ֥ים

ךָ מִיַּרְכְּתֵ֣י *צָפ֑וֹן* וַהֲבִאוֹתִ֖ךָ

קָּדִ֖ים וְהַ*צָּפֽוֹן* וְהַשַּׁ֗עַר

ו֙ דֶּ֣רֶךְ הַ*צָּפ֔וֹן* לֶחָצֵ֖ר הַ

הַשַּׁ֔עַר לַ*צָּפ֖וֹן* וְלַקָּדִ֑ים

אֶל־שַׁ֣עַר הַ*צָּפ֑וֹן* וּמָדַ֖ד כַּ

כֶּ֙תֶף֙ שַׁ֣עַר הַ*צָּפ֔וֹן* וּפְנֵיהֶ֖ם

הָ֙ דֶּ֣רֶךְ הַ*צָּפ֔וֹן* לַכֹּ֣הֲנִ֔ים שֹׁמְרֵ֖י

אֶחָד֙ דֶּ֣רֶךְ הַ*צָּפ֔וֹן* וּפֶ֥תַח אֶחָ֖ד

דֶּ֖רֶךְ דֶּ֣רֶךְ הַ*צָּפ֑וֹן* וַיְבִאֵ֣נִי

בִּנְיָ֖ן אֶל־הַ*צָּפֽוֹן* אֶל־פְּנֵי אֹ֨רֶךְ֙

מֵּאָ֔ה פֶּ֖תַח הַ*צָּפ֑וֹן* וְהָרֹ֖חַב

פִתְחֵיהֶ֖ם לַ*צָּפֽוֹן* וְהַלְּשָׁכ֥וֹת

אֲשֶׁר֙ דֶּ֣רֶךְ הַ*צָּפ֔וֹן* כְּאָרְכָ֖ן

ַ֗י לִֽשְׁכ֨וֹת הַ*צָּפ֜וֹן* לִֽשְׁכ֣וֹת הַדָּרוֹם֮

מָדַ֖ד ר֣וּחַ הַ*צָּפ֑וֹן* חֲמֵשׁ־מֵא֥וֹת קָנִ֛ים

דֶּֽרֶךְ־שַׁ֣עַר הַ*צָּפוֹן֮* אֶל פְּנֵ֣י הַ

בָּ֡א דֶּרֶךְ־שַׁ֨עַר *צָפ֜וֹן* לְהִֽשְׁתַּחֲוֹ֗ת יֵצֵא֙
