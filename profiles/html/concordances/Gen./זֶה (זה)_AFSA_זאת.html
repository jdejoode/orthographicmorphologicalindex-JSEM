בְּשָׂרִ֑י לְ*זֹאת֙* יִקָּרֵ֣א אִשָּׁ֔ה כִּ֥י

מֵאִ֖ישׁ לֻֽקֳחָה־*זֹּֽאת* עַל־כֵּן֙ יַֽעֲזָב־

לָאִשָּׁ֖ה מַה־*זֹּ֣את* עָשִׂ֑ית וַתֹּ֙אמֶר֙

נָּחָשׁ֮ כִּ֣י עָשִׂ֣יתָ *זֹּאת֒* אָר֤וּר אַתָּה֙ מִ

וַיֹּ֣אמֶר אֱלֹהִ֗ים *זֹ֤את* אֽוֹת־הַבְּרִית֙

אֱלֹהִ֖ים אֶל־נֹ֑חַ *זֹ֤את* אֽוֹת־הַבְּרִית֙

הָאָ֣רֶץ הַ*זֹּ֑את* וַיִּ֤בֶן שָׁם֙

אָמְר֖וּ אִשְׁת֣וֹ *זֹ֑את* וְהָרְג֥וּ אֹת

וַיֹּ֕אמֶר מַה־*זֹּ֖את* עָשִׂ֣יתָ לִ֑י

הָאָ֥רֶץ הַ*זֹּ֖את* לְרִשְׁתָֽה

הָאָ֣רֶץ הַ*זֹּ֔את* מִנְּהַ֣ר מִצְרַ֔יִם

לְדֹרֹתָֽם *זֹ֣את* בְּרִיתִ֞י אֲשֶׁ֣ר

הָעִ֨יר הַ*זֹּ֧את* קְרֹבָ֛ה לָנ֥וּס

כַּפַּ֖י עָשִׂ֥יתִי *זֹֽאת* וַיֹּאמֶר֩ אֵלָ֨י

לְבָבְךָ֙ עָשִׂ֣יתָ *זֹּ֔את* וָאֶחְשֹׂ֧ךְ גַּם־

הָאָמָ֥ה הַ*זֹּ֖את* וְאֶת־בְּנ

הָאָמָ֣ה הַ*זֹּ֔את* עִם־בְּנִ֖י

הַבְּאֵ֥ר הַ*זֹּֽאת* עַל־כֵּ֗ן קָרָ֛א

הָאָ֣רֶץ הַ*זֹּ֑את* הֶֽהָשֵׁ֤ב אָשִׁיב֙

הָאָ֣רֶץ הַ*זֹּ֑את* ה֗וּא יִשְׁלַ֤ח מַלְאָכ

מִשְּׁבֻעָתִ֖י *זֹ֑את* רַ֣ק אֶת־בְּנ

בָּאָ֣רֶץ הַ*זֹּ֔את* וְאֶֽהְיֶ֥ה עִמ

יֹּ֣אמֶר אֲבִימֶ֔לֶךְ מַה־*זֹּ֖את* עָשִׂ֣יתָ לָ֑נוּ

הָאֲדָמָ֖ה הַ*זֹּ֑את* כִּ֚י לֹ֣א אֶֽעֱזָב

הָאֶ֣בֶן הַ*זֹּ֗את* אֲשֶׁר־שַׂ֙מְתִּי֙ מַצֵּבָ֔ה

אֶל־לָבָ֗ן מַה־*זֹּאת֙* עָשִׂ֣יתָ לִ֔י

בְּכִירָֽה מַלֵּ֖א שְׁבֻ֣עַ *זֹ֑את* וְנִתְּנָ֨ה ל

ְךָ֜ גַּם־אֶת־*זֹ֗את* בַּעֲבֹדָה֙ אֲשֶׁ֣ר

וַיְמַלֵּ֖א שְׁבֻ֣עַ *זֹ֑את* וַיִּתֶּן־ל

הָאָ֣רֶץ הַ*זֹּ֔את* וְשׁ֖וּב אֶל־

הַמַּצֵּבָ֥ה הַ*זֹּ֖את* לְרָעָֽה אֱלֹהֵ֨י

הַיַּלְדָּ֥ה הַ*זֹּ֖את* לְאִשָּֽׁה וְ

ָֽנוּ אַךְ־בְּ*זֹ֖את* נֵא֣וֹת לָכֶ֑ם

ָהֶֽם אַךְ־בְּ֠*זֹאת* יֵאֹ֨תוּ לָ֤נוּ

הֶ֔ם וַיֹּאמְר֖וּ *זֹ֣את* מָצָ֑אנוּ הַכֶּר־נָ֗א

הַגְּדֹלָה֙ הַ*זֹּ֔את* וְחָטָ֖אתִי לֵֽ

ְךָ֖ אֶת־כָּל־*זֹ֑את* אֵין־נָב֥וֹן וְ

מְרַגְּלִ֥ים אַתֶּֽם בְּ*זֹ֖את* תִּבָּחֵ֑נוּ חֵ֤י פַרְעֹה֙

יּ֣וֹם הַשְּׁלִישִׁ֔י *זֹ֥את* עֲשׂ֖וּ וִֽחְי֑וּ

הַצָּרָ֖ה הַ*זֹּֽאת* וַיַּעַן֩ רְאוּבֵ֨ן

לֵאמֹ֔ר מַה־*זֹּ֛את* עָשָׂ֥ה אֱלֹהִ֖ים ל

הָאָ֔רֶץ בְּ*זֹ֣את* אֵדַ֔ע כִּ֥י כֵנִ֖ים

אִם־כֵּ֣ן ׀ אֵפוֹא֮ *זֹ֣את* עֲשׂוּ֒ קְח֞וּ מִ

הַמִּנְחָ֣ה הַ*זֹּ֔את* וּמִשְׁנֶה כֶּ֛סֶף

ִ֔י מֵעֲשׂ֖וֹת *זֹ֑את* הָאִ֡ישׁ אֲשֶׁר֩

אֶל־אַחֶ֖יךָ *זֹ֣את* עֲשׂ֑וּ טַֽעֲנוּ֙ אֶת־

וְאַתָּ֥ה צֻוֵּ֖יתָה *זֹ֣את* עֲשׂ֑וּ קְחוּ־ל

ו שָׁלַ֤ח כְּ*זֹאת֙* עֲשָׂרָ֣ה חֲמֹרִ֔ים נֹשְׂאִ֖ים

הָאָ֧רֶץ הַ*זֹּ֛את* לְזַרְעֲךָ֥

שְׁנֵ֣ים עָשָׂ֑ר וְ֠*זֹאת* אֲשֶׁר־דִּבֶּ֨ר ל
