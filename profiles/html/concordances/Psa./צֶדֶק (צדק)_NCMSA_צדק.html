סֶֽלָה זִבְח֥וּ זִבְחֵי־*צֶ֑דֶק* וּ֝בִטְח֗וּ אֶל־

לְ֝כִסֵּ֗א שׁוֹפֵ֥ט *צֶֽדֶק* גָּעַ֣רְתָּ ג֭וֹיִם אִבַּ֣דְתָּ

יִשְׁפֹּֽט־תֵּבֵ֥ל בְּ*צֶ֑דֶק* יָדִ֥ין לְ֝אֻמִּ֗ים בְּ

תָּ֭מִים וּפֹעֵ֥ל *צֶ֑דֶק* וְדֹבֵ֥ר אֱ֝מֶ֗ת

שִׁמְעָ֤ה יְהוָ֨ה ׀ *צֶ֗דֶק* הַקְשִׁ֥יבָה רִנָּת

הֶֽם אֲנִ֗י בְּ֭*צֶדֶק* אֶחֱזֶ֣ה פָנֶ֑יךָ

ֵ֥נִי בְמַעְגְּלֵי־*צֶ֝֗דֶק* לְמַ֣עַן שְׁמֽוֹ

מֵעָֽי בִּשַּׂ֤רְתִּי *צֶ֨דֶק ׀* בְּקָ֘הָ֤ל רָ֗ב

אֱ֭מֶת וְעַנְוָה־*צֶ֑דֶק* וְתוֹרְךָ֖

מַלְכוּתֶֽךָ אָהַ֣בְתָּ *צֶּדֶק֮* וַתִּשְׂנָ֫א רֶ֥שַׁע

עַל קַצְוֵי־אֶ֑רֶץ *צֶ֝֗דֶק* מָלְאָ֥ה יְמִינֶֽךָ

אָ֤ז תַּחְפֹּ֣ץ זִבְחֵי־*צֶ֭דֶק* עוֹלָ֣ה וְכָלִ֑יל

שֶׁ֓קֶר ׀ מִדַּבֵּ֖ר *צֶ֣דֶק* סֶֽלָה אָהַ֥בְתָּ כָֽל־

הַֽאֻמְנָ֗ם אֵ֣לֶם *צֶ֭דֶק* תְּדַבֵּר֑וּן מֵישָׁרִ֥ים

ֶֽךָ נ֤וֹרָא֨וֹת ׀ בְּ*צֶ֣דֶק* תַּ֭עֲנֵנוּ אֱלֹהֵ֣י

עַמְךָ֣ בְ*צֶ֑דֶק* וַעֲנִיֶּ֥יךָ

וֶאֱמֶ֥ת נִפְגָּ֑שׁוּ *צֶ֖דֶק* וְשָׁל֣וֹם נָשָֽׁקוּ

אֶ֣רֶץ תִּצְמָ֑ח וְ֝*צֶ֗דֶק* מִשָּׁמַ֥יִם נִשְׁקָֽף

תִּתֵּ֥ן יְבוּלָֽה *צֶ֭דֶק* לְפָנָ֣יו

תָּר֥וּם יְמִינֶֽךָ *צֶ֣דֶק* וּ֭מִשְׁפָּט מְכ֣וֹן

יַעֲזֹֽב כִּֽי־עַד־*צֶ֭דֶק* יָשׁ֣וּב מִשְׁפָּ֑ט וְ֝

יִשְׁפֹּֽט־תֵּבֵ֥ל בְּ*צֶ֑דֶק* וְ֝עַמִּ֗ים בֶּ

עֲרָפֶ֣ל סְבִיבָ֑יו *צֶ֥דֶק* וּ֝מִשְׁפָּ֗ט מְכ֣וֹן

יִשְׁפֹּֽט־תֵּבֵ֥ל בְּ*צֶ֑דֶק* וְ֝עַמִּ֗ים בְּ

לִ֥י שַׁעֲרֵי־*צֶ֑דֶק* אָֽבֹא בָ֝ם

יָדַ֣עְתִּי יְ֭הוָה כִּי־*צֶ֣דֶק* מִשְׁפָּטֶ֑יךָ וֶ֝

עָ֭שִׂיתִי מִשְׁפָּ֣ט וָ*צֶ֑דֶק* בַּל־תַּ֝נִּיחֵ֗נִי

שָׁכָֽחְתִּי צִדְקָתְךָ֣ *צֶ֣דֶק* לְעוֹלָ֑ם וְֽ

כָל־מִצְוֹתֶ֣יךָ *צֶּֽדֶק* תְּהִֽי־יָדְךָ֥

כֹּהֲנֶ֥יךָ יִלְבְּשׁוּ־*צֶ֑דֶק* וַחֲסִידֶ֥יךָ
